                    import pandas as pd

# Função para remover outliers com base no método do IQR
def remove_outliers(df, colunas):
    for coluna in colunas:
        Q1 = df[coluna].quantile(0.25)
        Q3 = df[coluna].quantile(0.75)
        IQR = Q3 - Q1
        
        # Limites inferior e superior para identificar outliers
        limite_inferior = Q1 - 1.5 * IQR
        limite_superior = Q3 + 1.5 * IQR
        
        # Filtra os dados removendo os outliers
        df = df[(df[coluna] >= limite_inferior) & (df[coluna] <= limite_superior)]
    
    return df

# Exemplo de uso
# Supondo que você tenha um DataFrame 'df' com uma ou mais colunas numéricas
df = pd.DataFrame({
    'A': [1, 2, 3, 4, 100, 6, 7, 8, 9],
    'B': [10, 11, 12, 13, 1000, 15, 16, 17, 18]
})

# Colunas que você deseja verificar e remover outliers
colunas_para_checar = ['A', 'B']

# Remover outliers
df_sem_outliers = remove_outliers(df, colunas_para_checar)

print(df_sem_outliers)
